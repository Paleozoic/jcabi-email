54  /**
55   * Envelope.
56   *
57   * <p>The best way is to use {@link com.jcabi.email.Envelope.MIME}, but
58   * you can easily create your own implementation of it.
59   *
60   * <p>It is recommended to always wrap your envelope into
61   * {@link com.jcabi.email.Envelope.Safe}.
62   *
63   * @author Yegor Bugayenko (yegor@teamed.io)
64   * @version $Id: a1f5c0329b2d3b009e4301aaf21c9773a180e42f $
65   * @since 1.0
66   */
67  @Immutable
68  public interface Envelope {
69  
70      /**
71       * Empty (always returns an empty MIME message).
72       * @since 1.5
73       */
74      Envelope EMPTY = new Envelope() {
75          @Override
76          public Message unwrap() {
77              return new MimeMessage(
78                  Session.getDefaultInstance(new Properties())
79              );
80          }
81      };
82  
83      /**
84       * Get a message out of it.
85       * @return Message to send
86       * @throws IOException If fails
87       */
88      Message unwrap() throws IOException;
89  
90      /**
91       * Default envelope.
92       */
93      @Immutable
94      @ToString
95      @EqualsAndHashCode(of = "encs")
96      @Loggable(Loggable.DEBUG)
97      final class MIME implements Envelope {
98          /**
99           * List of stamps.
100          */
101         private final transient Array<Stamp> stamps;
102         /**
103          * List of enclosures.
104          */
105         private final transient Array<Enclosure> encs;
106         /**
107          * Ctor.
108          * @since 1.3
109          */
110         public MIME() {
111             this(new Array<Stamp>(), new Array<Enclosure>());
112         }
113         /**
114          * Ctor.
115          * @param env Original envelope
116          * @since 1.5
117          */
118         public MIME(final Envelope env) {
119             this(
120                 Envelope.MIME.class.cast(env).stamps,
121                 Envelope.MIME.class.cast(env).encs
122             );
123         }
124         /**
125          * Ctor.
126          * @param stmps Stamps
127          * @param list List of enclosures
128          */
129         public MIME(final Iterable<Stamp> stmps,
130             final Iterable<Enclosure> list) {
131             this.stamps = new Array<>(stmps);
132             this.encs = new Array<>(list);
133         }
134         @Override
135         public Message unwrap() throws IOException {
136             final Message msg = Envelope.EMPTY.unwrap();
137             final Multipart multi = new MimeMultipart("alternative");
138             try {
139                 for (final Enclosure enc : this.encs) {
140                     multi.addBodyPart(enc.part());
141                 }
142                 for (final Stamp stamp : this.stamps) {
143                     stamp.attach(msg);
144                 }
145                 msg.setContent(multi);
146             } catch (final MessagingException ex) {
147                 throw new IOException(ex);
148             }
149             return msg;
150         }
151         /**
152          * With this stamp.
153          * @param stamp Stamp
154          * @return MIME envelope
155          * @since 1.3
156          */
157         public Envelope.MIME with(final Stamp stamp) {
158             return new Envelope.MIME(
159                 this.stamps.with(stamp),
160                 this.encs
161             );
162         }
163         /**
164          * With this enclosure.
165          * @param enc Enclosure
166          * @return MIME envelope
167          * @since 1.3
168          */
169         public Envelope.MIME with(final Enclosure enc) {
170             return new Envelope.MIME(
171                 this.stamps,
172                 this.encs.with(enc)
173             );
174         }
175     }

177     /**
178      * Strict envelope that fails if message is not valid.
179      * @since 1.3
180      */
181     @Immutable
182     @ToString
183     @EqualsAndHashCode(of = "origin")
184     @Loggable(Loggable.DEBUG)
185     final class Strict implements Envelope {
186         /**
187          * Origin env.
188          */
189         private final transient Envelope origin;
190         /**
191          * Ctor.
192          * @param env Envelope
193          */
194         public Strict(final Envelope env) {
195             this.origin = env;
196         }
197         @Override
198         public Message unwrap() throws IOException {
199             final Message msg = this.origin.unwrap();
200             try {
201                 if (msg.getAllRecipients() == null) {
202                     throw new IllegalStateException(
203                         "list of recipients is NULL"
204                     );
205                 }
206                 if (msg.getFrom() == null) {
207                     throw new IllegalStateException(
208                         "list of senders is NULL"
209                     );
210                 }
211                 if (msg.getSubject() == null) {
212                     throw new IllegalStateException(
213                         "subject is NULL"
214                     );
215                 }
216             } catch (final MessagingException ex) {
217                 throw new IOException(ex);
218             }
219             return msg;
220         }
221     }

223     /**
224      * Safe envelope that adds missing parts to the message.
225      * @since 1.3
226      */
227     @Immutable
228     @ToString
229     @EqualsAndHashCode(of = "origin")
230     @Loggable(Loggable.DEBUG)
231     final class Safe implements Envelope {
232         /**
233          * Origin env.
234          */
235         private final transient Envelope origin;
236         /**
237          * Ctor.
238          * @param env Envelope
239          */
240         public Safe(final Envelope env) {
241             this.origin = env;
242         }
243         @Override
244         public Message unwrap() throws IOException {
245             final Message msg = this.origin.unwrap();
246             try {
247                 if (msg.getAllRecipients() == null) {
248                     msg.addRecipient(
249                         Message.RecipientType.TO,
250                         new InternetAddress("to@example.com")
251                     );
252                     Logger.warn(this, "recipients were NULL, fake one set");
253                 }
254                 if (msg.getFrom() == null) {
255                     msg.setFrom(new InternetAddress("from@example.com"));
256                     Logger.warn(this, "senders were NULL, fake one set");
257                 }
258                 if (msg.getSubject() == null) {
259                     msg.setSubject(this.getClass().getCanonicalName());
260                     Logger.warn(this, "subject was NULL, fake one set");
261                 }
262             } catch (final MessagingException ex) {
263                 throw new IOException(ex);
264             }
265             return msg;
266         }
267     }

269     /**
270      * Envelope that always returns the same message (within one hour).
271      * @since 1.4
272      */
273     @Immutable
274     @ToString
275     @EqualsAndHashCode(of = "origin")
276     @Loggable(Loggable.DEBUG)
277     final class Constant implements Envelope {
278         /**
279          * Guava cache.
280          */
281         private static final Cache<Envelope, Message> CACHE =
282             CacheBuilder.newBuilder()
283                 .expireAfterWrite(1L, TimeUnit.HOURS)
284                 .build();
285         /**
286          * Origin env.
287          */
288         private final transient Envelope origin;
289         /**
290          * Ctor.
291          * @param env Envelope
292          */
293         public Constant(final Envelope env) {
294             this.origin = env;
295         }
296         @Override
297         public Message unwrap() throws IOException {
298             try {
299                 return Envelope.Constant.CACHE.get(
300                     this.origin,
301                     new Callable<Message>() {
302                         @Override
303                         public Message call() throws Exception {
304                             return Envelope.Constant.this.origin.unwrap();
305                         }
306                     }
307                 );
308             } catch (final ExecutionException ex) {
309                 throw new IOException(ex);
310             }
311         }
312     }

314     /**
315      * Envelope that adds DRAFT flag to the message.
316      * @since 1.7
317      */
318     @Immutable
319     @ToString
320     @EqualsAndHashCode(of = "env")
321     @Loggable(Loggable.DEBUG)
322     final class Draft implements Envelope {
323         /**
324          * Origin env.
325          */
326         private final transient Envelope env;
327         /**
328          * Ctor.
329          * @param origin Envelope
330          */
331         public Draft(final Envelope origin) {
332             this.env = origin;
333         }
334         @Override
335         public Message unwrap() throws IOException {
336             final Message msg = this.env.unwrap();
337             try {
338                 msg.setFlag(Flags.Flag.DRAFT, true);
339             } catch (final MessagingException ex) {
340                 throw new IOException(ex);
341             }
342             return msg;
343         }
344     }
